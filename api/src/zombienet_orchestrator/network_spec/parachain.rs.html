99    pub fn from_config(
100        config: &ParachainConfig,
101        relay_chain: Chain,
102    ) -> Result<ParachainSpec, OrchestratorError> {
103        let main_cmd = if let Some(cmd) = config.default_command() {
104            cmd
105        } else if let Some(first_node) = config.collators().first() {
106            let Some(cmd) = first_node.command() else {
107                return Err(OrchestratorError::InvalidConfig(format!("Parachain {}, either default_command or command in the first node needs to be set.", config.id())));
108            };
109
110            cmd
111        } else {
112            return Err(OrchestratorError::InvalidConfig(format!(
113                "Parachain {}, without nodes and default_command isn't set.",
114                config.id()
115            )));
116        };
117
118        // TODO: internally we use image as String
119        let main_image = config
120            .default_image()
121            .or(config.collators().first().and_then(|node| node.image()))
122            .map(|image| image.as_str().to_string());
123
124        let chain_spec = if config.is_cumulus_based() {
125            // we need a chain-spec
126            let chain_name = if let Some(chain_name) = config.chain() {
127                chain_name.as_str()
128            } else {
129                ""
130            };
131
132            let chain_spec_builder = if chain_name.is_empty() {
133                // if the chain don't have name use the unique_id for the name of the file
134                ChainSpec::new(
135                    config.unique_id().to_string(),
136                    Context::Para {
137                        relay_chain,
138                        para_id: config.id(),
139                    },
140                )
141            } else {
142                let chain_spec_file_name = if config.unique_id().contains('-') {
143                    &format!("{}-{}", chain_name, config.unique_id())
144                } else {
145                    chain_name
146                };
147                ChainSpec::new(
148                    chain_spec_file_name,
149                    Context::Para {
150                        relay_chain,
151                        para_id: config.id(),
152                    },
153                )
154            };
155            let chain_spec_builder = chain_spec_builder.set_chain_name(chain_name);
156
157            let replacements = HashMap::from([
158                ("disableBootnodes", "--disable-default-bootnode"),
159                ("mainCommand", main_cmd.as_str()),
160            ]);
161            let tmpl = if let Some(tmpl) = config.chain_spec_command() {
162                apply_replacements(tmpl, &replacements)
163            } else {
164                apply_replacements(DEFAULT_CHAIN_SPEC_TPL_COMMAND, &replacements)
165            };
166
167            let chain_spec = chain_spec_builder
168                .command(
169                    tmpl.as_str(),
170                    config.chain_spec_command_is_local(),
171                    config.chain_spec_command_output_path(),
172                )
173                .image(main_image.clone());
174
175            let chain_spec = if let Some(chain_spec_path) = config.chain_spec_path() {
176                chain_spec.asset_location(chain_spec_path.clone())
177            } else {
178                chain_spec
179            };
180
181            // add chain-spec runtime if present
182            let chain_spec = if let Some(chain_spec_runtime) = config.chain_spec_runtime() {
183                chain_spec.runtime(chain_spec_runtime.clone())
184            } else {
185                chain_spec
186            };
187
188            Some(chain_spec)
189        } else {
190            None
191        };
192
193        // build the `node_specs`
194        let chain_context = ChainDefaultContext {
195            default_command: config.default_command(),
196            default_image: config.default_image(),
197            default_resources: config.default_resources(),
198            default_db_snapshot: config.default_db_snapshot(),
199            default_args: config.default_args(),
200        };
201
202        // We want to track the errors for all the nodes and report them ones
203        let mut errs: Vec<OrchestratorError> = Default::default();
204        let mut collators: Vec<NodeSpec> = Default::default();
205
206        let mut nodes: Vec<NodeConfig> = config.collators().into_iter().cloned().collect();
207        nodes.extend(
208            config
209                .group_collators_configs()
210                .into_iter()
211                .flat_map(|node_group| node_group.expand_group_configs()),
212        );
213
214        let mut names = HashSet::new();
215        for node_config in nodes {
216            match NodeSpec::from_config(&node_config, &chain_context, true, config.is_evm_based()) {
217                Ok(mut node) => {
218                    let unique_name = generate_unique_node_name_from_names(node.name, &mut names);
219                    node.name = unique_name;
220                    collators.push(node)
221                },
222                Err(err) => errs.push(err),
223            }
224        }
225        // Precedence when deciding the genesis state artifact: path -> generator -> default command
226        let genesis_state = if let Some(path) = config.genesis_state_path() {
227            ParaArtifact::new(
228                ParaArtifactType::State,
229                ParaArtifactBuildOption::Path(path.to_string()),
230            )
231        } else if let Some(cmd) = config.genesis_state_generator() {
232            ParaArtifact::new(
233                ParaArtifactType::State,
234                ParaArtifactBuildOption::CommandWithCustomArgs(cmd.clone()),
235            )
236            .image(main_image.clone())
237        } else {
238            ParaArtifact::new(
239                ParaArtifactType::State,
240                ParaArtifactBuildOption::Command(main_cmd.as_str().into()),
241            )
242            .image(main_image.clone())
243        };
244
245        debug!(
246            "Parachain {} genesis state artifact: {:?}",
247            config.id(),
248            genesis_state
249        );
250
251        let genesis_wasm = if let Some(path) = config.genesis_wasm_path() {
252            ParaArtifact::new(
253                ParaArtifactType::Wasm,
254                ParaArtifactBuildOption::Path(path.to_string()),
255            )
256        } else {
257            let cmd = if let Some(cmd) = config.genesis_wasm_generator() {
258                cmd.as_str()
259            } else {
260                main_cmd.as_str()
261            };
262            ParaArtifact::new(
263                ParaArtifactType::Wasm,
264                ParaArtifactBuildOption::Command(cmd.into()),
265            )
266            .image(main_image.clone())
267        };
268
269        let para_spec = ParachainSpec {
270            id: config.id(),
271            // ensure unique id is set at this point, if not just set to the para_id
272            unique_id: if config.unique_id().is_empty() {
273                config.id().to_string()
274            } else {
275                config.unique_id().to_string()
276            },
277            default_command: config.default_command().cloned(),
278            default_image: config.default_image().cloned(),
279            default_resources: config.default_resources().cloned(),
280            default_db_snapshot: config.default_db_snapshot().cloned(),
281            wasm_override: config.wasm_override().cloned(),
282            default_args: config.default_args().into_iter().cloned().collect(),
283            chain_spec,
284            no_default_bootnodes: config.no_default_bootnodes(),
285            registration_strategy: config
286                .registration_strategy()
287                .unwrap_or(&RegistrationStrategy::InGenesis)
288                .clone(),
289            onboard_as_parachain: config.onboard_as_parachain(),
290            is_cumulus_based: config.is_cumulus_based(),
291            is_evm_based: config.is_evm_based(),
292            initial_balance: config.initial_balance(),
293            genesis_state,
294            genesis_wasm,
295            genesis_overrides: config.genesis_overrides().cloned(),
296            collators,
297            raw_spec_override: config.raw_spec_override().cloned(),
298            bootnodes_addresses: config.bootnodes_addresses().into_iter().cloned().collect(),
299            post_process_script: config.post_process_script().map(str::to_string),
300        };
301
302        Ok(para_spec)
303    }

309    pub fn get_genesis_config(&self) -> Result<ParaGenesisConfig<&PathBuf>, OrchestratorError> {
310        let genesis_config = ParaGenesisConfig {
311            state_path: self.genesis_state.artifact_path().ok_or(
312                OrchestratorError::InvariantError(
313                    "artifact path for state must be set at this point",
314                ),
315            )?,
316            wasm_path: self.genesis_wasm.artifact_path().ok_or(
317                OrchestratorError::InvariantError(
318                    "artifact path for wasm must be set at this point",
319                ),
320            )?,
321            id: self.id,
322            as_parachain: self.onboard_as_parachain,
323        };
324        Ok(genesis_config)
325    }

339    /// Build parachain chain-spec
340    ///
341    /// This function customize the chain-spec (if is possible) and build the raw version
342    /// of the chain-spec.
343    pub(crate) async fn build_chain_spec<'a, T>(
344        &mut self,
345        relay_chain_id: &str,
346        ns: &DynNamespace,
347        scoped_fs: &ScopedFilesystem<'a, T>,
348        post_process_script: Option<&str>,
349    ) -> Result<Option<PathBuf>, anyhow::Error>
350    where
351        T: FileSystem,
352    {
353        let cloned = self.clone();
354        let chain_spec_raw_path = if let Some(chain_spec) = self.chain_spec.as_mut() {
355            debug!("parachain chain-spec building!");
356            chain_spec.build(ns, scoped_fs).await?;
357            debug!("parachain chain-spec built!");
358
359            chain_spec
360                .customize_para(&cloned, relay_chain_id, scoped_fs)
361                .await?;
362            debug!("parachain chain-spec customized!");
363            // Run post-process script if configured for this parachain (run against plain spec before building raw)
364            if let Some(script_cmd) = post_process_script {
365                chain_spec
366                    .run_post_process_script(script_cmd, scoped_fs)
367                    .await?;
368            }
369
370            chain_spec
371                .build_raw(ns, scoped_fs, Some(relay_chain_id.try_into()?))
372                .await?;
373            debug!("parachain chain-spec raw built!");
374
375            // override wasm if needed
376            if let Some(ref wasm_override) = self.wasm_override {
377                chain_spec.override_code(scoped_fs, wasm_override).await?;
378            }
379
380            // override raw spec if needed
381            if let Some(ref raw_spec_override) = self.raw_spec_override {
382                chain_spec
383                    .override_raw_spec(scoped_fs, raw_spec_override)
384                    .await?;
385            }
386
387            let chain_spec_raw_path =
388                chain_spec
389                    .raw_path()
390                    .ok_or(OrchestratorError::InvariantError(
391                        "chain-spec raw path should be set now",
392                    ))?;
393
394            Some(chain_spec_raw_path.to_path_buf())
395        } else {
396            None
397        };
398        Ok(chain_spec_raw_path)
399    }
