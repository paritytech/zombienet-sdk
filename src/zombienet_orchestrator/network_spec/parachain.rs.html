97    pub fn from_config(
98        config: &ParachainConfig,
99        relay_chain: Chain,
100    ) -> Result<ParachainSpec, OrchestratorError> {
101        let main_cmd = if let Some(cmd) = config.default_command() {
102            cmd
103        } else if let Some(first_node) = config.collators().first() {
104            let Some(cmd) = first_node.command() else {
105                return Err(OrchestratorError::InvalidConfig(format!("Parachain {}, either default_command or command in the first node needs to be set.", config.id())));
106            };
107
108            cmd
109        } else {
110            return Err(OrchestratorError::InvalidConfig(format!(
111                "Parachain {}, without nodes and default_command isn't set.",
112                config.id()
113            )));
114        };
115
116        // TODO: internally we use image as String
117        let main_image = config
118            .default_image()
119            .or(config.collators().first().and_then(|node| node.image()))
120            .map(|image| image.as_str().to_string());
121
122        let chain_spec = if config.is_cumulus_based() {
123            // we need a chain-spec
124            let chain_name = if let Some(chain_name) = config.chain() {
125                chain_name.as_str()
126            } else {
127                ""
128            };
129
130            let chain_spec_builder = if chain_name.is_empty() {
131                // if the chain don't have name use the unique_id for the name of the file
132                ChainSpec::new(
133                    config.unique_id().to_string(),
134                    Context::Para {
135                        relay_chain,
136                        para_id: config.id(),
137                    },
138                )
139            } else {
140                let chain_spec_file_name = if config.unique_id().contains('-') {
141                    &format!("{}-{}", chain_name, config.unique_id())
142                } else {
143                    chain_name
144                };
145                ChainSpec::new(
146                    chain_spec_file_name,
147                    Context::Para {
148                        relay_chain,
149                        para_id: config.id(),
150                    },
151                )
152            };
153            let chain_spec_builder = chain_spec_builder.set_chain_name(chain_name);
154
155            let replacements = HashMap::from([
156                ("disableBootnodes", "--disable-default-bootnode"),
157                ("mainCommand", main_cmd.as_str()),
158            ]);
159            let tmpl = if let Some(tmpl) = config.chain_spec_command() {
160                apply_replacements(tmpl, &replacements)
161            } else {
162                apply_replacements(DEFAULT_CHAIN_SPEC_TPL_COMMAND, &replacements)
163            };
164
165            let chain_spec = chain_spec_builder
166                .command(
167                    tmpl.as_str(),
168                    config.chain_spec_command_is_local(),
169                    config.chain_spec_command_output_path(),
170                )
171                .image(main_image.clone());
172
173            let chain_spec = if let Some(chain_spec_path) = config.chain_spec_path() {
174                chain_spec.asset_location(chain_spec_path.clone())
175            } else {
176                chain_spec
177            };
178
179            // add chain-spec runtime if present
180            let chain_spec = if let Some(chain_spec_runtime) = config.chain_spec_runtime() {
181                chain_spec.runtime(chain_spec_runtime.clone())
182            } else {
183                chain_spec
184            };
185
186            Some(chain_spec)
187        } else {
188            None
189        };
190
191        // build the `node_specs`
192        let chain_context = ChainDefaultContext {
193            default_command: config.default_command(),
194            default_image: config.default_image(),
195            default_resources: config.default_resources(),
196            default_db_snapshot: config.default_db_snapshot(),
197            default_args: config.default_args(),
198        };
199
200        // We want to track the errors for all the nodes and report them ones
201        let mut errs: Vec<OrchestratorError> = Default::default();
202        let mut collators: Vec<NodeSpec> = Default::default();
203
204        let mut nodes: Vec<NodeConfig> = config.collators().into_iter().cloned().collect();
205        nodes.extend(
206            config
207                .group_collators_configs()
208                .into_iter()
209                .flat_map(|node_group| node_group.expand_group_configs()),
210        );
211
212        let mut names = HashSet::new();
213        for node_config in nodes {
214            match NodeSpec::from_config(&node_config, &chain_context, true) {
215                Ok(mut node) => {
216                    let unique_name = generate_unique_node_name_from_names(node.name, &mut names);
217                    node.name = unique_name;
218                    collators.push(node)
219                },
220                Err(err) => errs.push(err),
221            }
222        }
223        let genesis_state = if let Some(path) = config.genesis_state_path() {
224            ParaArtifact::new(
225                ParaArtifactType::State,
226                ParaArtifactBuildOption::Path(path.to_string()),
227            )
228        } else {
229            let cmd = if let Some(cmd) = config.genesis_state_generator() {
230                cmd.cmd()
231            } else {
232                main_cmd
233            };
234            ParaArtifact::new(
235                ParaArtifactType::State,
236                ParaArtifactBuildOption::Command(cmd.as_str().into()),
237            )
238            .image(main_image.clone())
239        };
240
241        let genesis_wasm = if let Some(path) = config.genesis_wasm_path() {
242            ParaArtifact::new(
243                ParaArtifactType::Wasm,
244                ParaArtifactBuildOption::Path(path.to_string()),
245            )
246        } else {
247            let cmd = if let Some(cmd) = config.genesis_wasm_generator() {
248                cmd.as_str()
249            } else {
250                main_cmd.as_str()
251            };
252            ParaArtifact::new(
253                ParaArtifactType::Wasm,
254                ParaArtifactBuildOption::Command(cmd.into()),
255            )
256            .image(main_image.clone())
257        };
258
259        let para_spec = ParachainSpec {
260            id: config.id(),
261            // ensure unique id is set at this point, if not just set to the para_id
262            unique_id: if config.unique_id().is_empty() {
263                config.id().to_string()
264            } else {
265                config.unique_id().to_string()
266            },
267            default_command: config.default_command().cloned(),
268            default_image: config.default_image().cloned(),
269            default_resources: config.default_resources().cloned(),
270            default_db_snapshot: config.default_db_snapshot().cloned(),
271            wasm_override: config.wasm_override().cloned(),
272            default_args: config.default_args().into_iter().cloned().collect(),
273            chain_spec,
274            no_default_bootnodes: config.no_default_bootnodes(),
275            registration_strategy: config
276                .registration_strategy()
277                .unwrap_or(&RegistrationStrategy::InGenesis)
278                .clone(),
279            onboard_as_parachain: config.onboard_as_parachain(),
280            is_cumulus_based: config.is_cumulus_based(),
281            is_evm_based: config.is_evm_based(),
282            initial_balance: config.initial_balance(),
283            genesis_state,
284            genesis_wasm,
285            genesis_overrides: config.genesis_overrides().cloned(),
286            collators,
287            raw_spec_override: config.raw_spec_override().cloned(),
288            bootnodes_addresses: config.bootnodes_addresses().into_iter().cloned().collect(),
289        };
290
291        Ok(para_spec)
292    }

298    pub fn get_genesis_config(&self) -> Result<ParaGenesisConfig<&PathBuf>, OrchestratorError> {
299        let genesis_config = ParaGenesisConfig {
300            state_path: self.genesis_state.artifact_path().ok_or(
301                OrchestratorError::InvariantError(
302                    "artifact path for state must be set at this point",
303                ),
304            )?,
305            wasm_path: self.genesis_wasm.artifact_path().ok_or(
306                OrchestratorError::InvariantError(
307                    "artifact path for wasm must be set at this point",
308                ),
309            )?,
310            id: self.id,
311            as_parachain: self.onboard_as_parachain,
312        };
313        Ok(genesis_config)
314    }

328    /// Build parachain chain-spec
329    ///
330    /// This function customize the chain-spec (if is possible) and build the raw version
331    /// of the chain-spec.
332    pub(crate) async fn build_chain_spec<'a, T>(
333        &mut self,
334        relay_chain_id: &str,
335        ns: &DynNamespace,
336        scoped_fs: &ScopedFilesystem<'a, T>,
337    ) -> Result<Option<PathBuf>, anyhow::Error>
338    where
339        T: FileSystem,
340    {
341        let cloned = self.clone();
342        let chain_spec_raw_path = if let Some(chain_spec) = self.chain_spec.as_mut() {
343            debug!("parachain chain-spec building!");
344            chain_spec.build(ns, scoped_fs).await?;
345            debug!("parachain chain-spec built!");
346
347            chain_spec
348                .customize_para(&cloned, relay_chain_id, scoped_fs)
349                .await?;
350            debug!("parachain chain-spec customized!");
351            chain_spec
352                .build_raw(ns, scoped_fs, Some(relay_chain_id.try_into()?))
353                .await?;
354            debug!("parachain chain-spec raw built!");
355
356            // override wasm if needed
357            if let Some(ref wasm_override) = self.wasm_override {
358                chain_spec.override_code(scoped_fs, wasm_override).await?;
359            }
360
361            // override raw spec if needed
362            if let Some(ref raw_spec_override) = self.raw_spec_override {
363                chain_spec
364                    .override_raw_spec(scoped_fs, raw_spec_override)
365                    .await?;
366            }
367
368            let chain_spec_raw_path =
369                chain_spec
370                    .raw_path()
371                    .ok_or(OrchestratorError::InvariantError(
372                        "chain-spec raw path should be set now",
373                    ))?;
374
375            Some(chain_spec_raw_path.to_path_buf())
376        } else {
377            None
378        };
379        Ok(chain_spec_raw_path)
380    }
