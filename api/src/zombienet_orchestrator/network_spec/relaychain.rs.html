22/// A relaychain configuration spec
23#[derive(Debug, Clone, Serialize, Deserialize)]
24pub struct RelaychainSpec {
25    /// Chain to use (e.g. rococo-local).
26    pub(crate) chain: Chain,
27
28    /// Default command to run the node. Can be overridden on each node.
29    pub(crate) default_command: Option<Command>,
30
31    /// Default image to use (only podman/k8s). Can be overridden on each node.
32    pub(crate) default_image: Option<Image>,
33
34    /// Default resources. Can be overridden on each node.
35    pub(crate) default_resources: Option<Resources>,
36
37    /// Default database snapshot. Can be overridden on each node.
38    pub(crate) default_db_snapshot: Option<AssetLocation>,
39
40    /// Default arguments to use in nodes. Can be overridden on each node.
41    pub(crate) default_args: Vec<Arg>,
42
43    // chain_spec_path: Option<AssetLocation>,
44    pub(crate) chain_spec: ChainSpec,
45
46    /// Set the count of nominators to generator (used with PoS networks).
47    pub(crate) random_nominators_count: u32,
48
49    /// Set the max nominators value (used with PoS networks).
50    pub(crate) max_nominations: u8,
51
52    /// Genesis overrides as JSON value.
53    pub(crate) runtime_genesis_patch: Option<serde_json::Value>,
54
55    /// Wasm override path/url to use.
56    pub(crate) wasm_override: Option<AssetLocation>,
57
58    /// Nodes to run.
59    pub(crate) nodes: Vec<NodeSpec>,
60
61    /// Raw chain-spec override path, url or inline json to use.
62    pub(crate) raw_spec_override: Option<JsonOverrides>,
63
64    /// Optional post-process script configured for this relaychain
65    pub(crate) post_process_script: Option<String>,
66}

68impl RelaychainSpec {
69    pub fn from_config(config: &RelaychainConfig) -> Result<RelaychainSpec, OrchestratorError> {
70        // Relaychain main command to use, in order:
71        // set as `default_command` or
72        // use the command of the first node.
73        // If non of those is set, return an error.
74        let main_cmd = config
75            .default_command()
76            .or(config.nodes().first().and_then(|node| node.command()))
77            .ok_or(OrchestratorError::InvalidConfig(
78                "Relaychain, either default_command or first node with a command needs to be set."
79                    .to_string(),
80            ))?;
81
82        // TODO: internally we use image as String
83        let main_image = config
84            .default_image()
85            .or(config.nodes().first().and_then(|node| node.image()))
86            .map(|image| image.as_str().to_string());
87
88        let replacements = HashMap::from([
89            ("disableBootnodes", "--disable-default-bootnode"),
90            ("mainCommand", main_cmd.as_str()),
91        ]);
92        let tmpl = if let Some(tmpl) = config.chain_spec_command() {
93            apply_replacements(tmpl, &replacements)
94        } else {
95            apply_replacements(DEFAULT_CHAIN_SPEC_TPL_COMMAND, &replacements)
96        };
97
98        let chain_spec = ChainSpec::new(config.chain().as_str(), Context::Relay)
99            .set_chain_name(config.chain().as_str())
100            .command(
101                tmpl.as_str(),
102                config.chain_spec_command_is_local(),
103                config.chain_spec_command_output_path(),
104            )
105            .image(main_image.clone());
106
107        // Add asset location if present
108        let chain_spec = if let Some(chain_spec_path) = config.chain_spec_path() {
109            chain_spec.asset_location(chain_spec_path.clone())
110        } else {
111            chain_spec
112        };
113
114        // add chain-spec runtime  if present
115        let chain_spec = if let Some(chain_spec_runtime) = config.chain_spec_runtime() {
116            chain_spec.runtime(chain_spec_runtime.clone())
117        } else {
118            chain_spec
119        };
120
121        // build the `node_specs`
122        let chain_context = ChainDefaultContext {
123            default_command: config.default_command(),
124            default_image: config.default_image(),
125            default_resources: config.default_resources(),
126            default_db_snapshot: config.default_db_snapshot(),
127            default_args: config.default_args(),
128        };
129
130        let mut nodes: Vec<NodeConfig> = config.nodes().into_iter().cloned().collect();
131        nodes.extend(
132            config
133                .group_node_configs()
134                .into_iter()
135                .flat_map(|node_group| node_group.expand_group_configs()),
136        );
137
138        let mut names = HashSet::new();
139        let (nodes, mut errs) = nodes
140            .iter()
141            .map(|node_config| NodeSpec::from_config(node_config, &chain_context, false, false))
142            .fold((vec![], vec![]), |(mut nodes, mut errs), result| {
143                match result {
144                    Ok(mut node) => {
145                        let unique_name =
146                            generate_unique_node_name_from_names(node.name, &mut names);
147                        node.name = unique_name;
148                        nodes.push(node);
149                    },
150                    Err(err) => errs.push(err),
151                }
152                (nodes, errs)
153            });
154
155        if !errs.is_empty() {
156            // TODO: merge errs, maybe return something like Result<Sometype, Vec<OrchestratorError>>
157            return Err(errs.swap_remove(0));
158        }
159
160        Ok(RelaychainSpec {
161            chain: config.chain().clone(),
162            default_command: config.default_command().cloned(),
163            default_image: config.default_image().cloned(),
164            default_resources: config.default_resources().cloned(),
165            default_db_snapshot: config.default_db_snapshot().cloned(),
166            wasm_override: config.wasm_override().cloned(),
167            default_args: config.default_args().into_iter().cloned().collect(),
168            chain_spec,
169            random_nominators_count: config.random_nominators_count().unwrap_or(0),
170            max_nominations: config.max_nominations().unwrap_or(24),
171            runtime_genesis_patch: config.runtime_genesis_patch().cloned(),
172            nodes,
173            raw_spec_override: config.raw_spec_override().cloned(),
174            post_process_script: config.post_process_script().map(str::to_string),
175        })
176    }
177
178    pub fn chain_spec(&self) -> &ChainSpec {
179        &self.chain_spec
180    }
181
182    pub fn chain_spec_mut(&mut self) -> &mut ChainSpec {
183        &mut self.chain_spec
184    }
