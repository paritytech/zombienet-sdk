80    pub(super) async fn new(
81        options: KubernetesNodeOptions<'_, FS>,
82    ) -> Result<Arc<Self>, ProviderError> {
83        let image = options.image.ok_or_else(|| {
84            ProviderError::MissingNodeInfo(options.name.to_string(), "missing image".to_string())
85        })?;
86
87        let filesystem = options.filesystem.clone();
88
89        let base_dir =
90            PathBuf::from_iter([options.namespace_base_dir, &PathBuf::from(options.name)]);
91        filesystem.create_dir_all(&base_dir).await?;
92
93        let base_dir_raw = base_dir.to_string_lossy();
94        let config_dir = PathBuf::from(format!("{base_dir_raw}{NODE_CONFIG_DIR}"));
95        let data_dir = PathBuf::from(format!("{base_dir_raw}{NODE_DATA_DIR}"));
96        let relay_data_dir = PathBuf::from(format!("{base_dir_raw}{NODE_RELAY_DATA_DIR}"));
97        let scripts_dir = PathBuf::from(format!("{base_dir_raw}{NODE_SCRIPTS_DIR}"));
98        let log_path = base_dir.join("node.log");
99
100        try_join!(
101            filesystem.create_dir(&config_dir),
102            filesystem.create_dir(&data_dir),
103            filesystem.create_dir(&relay_data_dir),
104            filesystem.create_dir(&scripts_dir),
105        )?;
106
107        let node = Arc::new(KubernetesNode {
108            namespace: options.namespace.clone(),
109            name: options.name.to_string(),
110            image: image.to_string(),
111            program: options.program.to_string(),
112            args: options.args.to_vec(),
113            env: options.env.to_vec(),
114            resources: options.resources.cloned(),
115            base_dir,
116            config_dir,
117            data_dir,
118            relay_data_dir,
119            scripts_dir,
120            log_path,
121            filesystem: filesystem.clone(),
122            k8s_client: options.k8s_client.clone(),
123            http_client: reqwest::Client::new(),
124            port_fwds: Default::default(),
125        });
126
127        node.initialize_k8s().await?;
128
129        if let Some(db_snap) = options.db_snapshot {
130            node.initialize_db_snapshot(db_snap).await?;
131        }
132
133        node.initialize_startup_files(options.startup_files).await?;
134
135        node.start().await?;
136
137        Ok(node)
138    }

140    async fn initialize_k8s(&self) -> Result<(), ProviderError> {
141        let labels = BTreeMap::from([
142            (
143                "app.kubernetes.io/name".to_string(),
144                self.name().to_string(),
145            ),
146            (
147                "x-infra-instance".to_string(),
148                env::var("X_INFRA_INSTANCE").unwrap_or("ondemand".to_string()),
149            ),
150        ]);
151
152        // Create pod
153        let pod_spec = PodSpecBuilder::build(
154            &self.name,
155            &self.image,
156            self.resources.as_ref(),
157            &self.program,
158            &self.args,
159            &self.env,
160        );
161
162        let manifest = self
163            .k8s_client
164            .create_pod(&self.namespace_name(), &self.name, pod_spec, labels.clone())
165            .await
166            .map_err(|err| ProviderError::NodeSpawningFailed(self.name.clone(), err.into()))?;
167
168        let serialized_manifest = serde_yaml::to_string(&manifest)
169            .map_err(|err| ProviderError::NodeSpawningFailed(self.name.to_string(), err.into()))?;
170
171        let dest_path = PathBuf::from_iter([
172            &self.base_dir,
173            &PathBuf::from(format!("{}_manifest.yaml", &self.name)),
174        ]);
175
176        self.filesystem
177            .write(dest_path, serialized_manifest)
178            .await
179            .map_err(|err| ProviderError::NodeSpawningFailed(self.name.to_string(), err.into()))?;
180
181        // Create service for pod
182        let service_spec = ServiceSpec {
183            selector: Some(labels.clone()),
184            ports: Some(vec![
185                ServicePort {
186                    port: P2P_PORT.into(),
187                    name: Some("p2p".into()),
188                    ..Default::default()
189                },
190                ServicePort {
191                    port: RPC_WS_PORT.into(),
192                    name: Some("rpc".into()),
193                    ..Default::default()
194                },
195                ServicePort {
196                    port: RPC_HTTP_PORT.into(),
197                    name: Some("rpc-http".into()),
198                    ..Default::default()
199                },
200                ServicePort {
201                    port: PROMETHEUS_PORT.into(),
202                    name: Some("prom".into()),
203                    ..Default::default()
204                },
205            ]),
206            ..Default::default()
207        };
208
209        let service_manifest = self
210            .k8s_client
211            .create_service(&self.namespace_name(), &self.name, service_spec, labels)
212            .await
213            .map_err(|err| ProviderError::FileServerSetupError(err.into()))?;
214
215        let serialized_service_manifest = serde_yaml::to_string(&service_manifest)
216            .map_err(|err| ProviderError::FileServerSetupError(err.into()))?;
217
218        let service_dest_path = PathBuf::from_iter([
219            &self.base_dir,
220            &PathBuf::from(format!("{}_svc_manifest.yaml", &self.name)),
221        ]);
222
223        self.filesystem
224            .write(service_dest_path, serialized_service_manifest)
225            .await?;
226
227        Ok(())
228    }

230    async fn initialize_db_snapshot(
231        &self,
232        db_snapshot: &AssetLocation,
233    ) -> Result<(), ProviderError> {
234        trace!("snap: {db_snapshot}");
235        let url_of_snap = match db_snapshot {
236            AssetLocation::Url(location) => location.clone(),
237            AssetLocation::FilePath(filepath) => {
238                let (url, _) = self.upload_to_fileserver(filepath).await?;
239                url
240            },
241        };
242
243        // we need to get the snapshot from a public access
244        // and extract to /data
245        let opts = RunCommandOptions::new("mkdir").args([
246            "-p",
247            "/data/",
248            "&&",
249            "mkdir",
250            "-p",
251            "/relay-data/",
252            "&&",
253            // Use our version of curl
254            "/cfg/curl",
255            url_of_snap.as_ref(),
256            "--output",
257            "/data/db.tgz",
258            "&&",
259            "cd",
260            "/",
261            "&&",
262            "tar",
263            "--skip-old-files",
264            "-xzvf",
265            "/data/db.tgz",
266        ]);
267
268        trace!("cmd opts: {:#?}", opts);
269        let _ = self.run_command(opts).await?;
270
271        Ok(())
272    }

355    async fn upload_to_fileserver(&self, location: &Path) -> Result<(Url, String), ProviderError> {
356        let file_name = if let Some(name) = location.file_name() {
357            name.to_string_lossy()
358        } else {
359            "unnamed".into()
360        };
361
362        let data = self.filesystem.read(location).await?;
363        let content_hashed = hex::encode(sha2::Sha256::digest(&data));
364        let req = self
365            .http_client
366            .head(format!(
367                "http://{}/{content_hashed}__{file_name}",
368                self.file_server_local_host().await?
369            ))
370            .build()
371            .map_err(|err| {
372                ProviderError::UploadFile(location.to_string_lossy().to_string(), err.into())
373            })?;
374
375        let url = req.url().clone();
376        let res = self.http_client.execute(req).await.map_err(|err| {
377            ProviderError::UploadFile(location.to_string_lossy().to_string(), err.into())
378        })?;
379
380        if res.status() != reqwest::StatusCode::OK {
381            // we need to upload the file
382            self.http_client
383                .post(url.as_ref())
384                .body(data)
385                .send()
386                .await
387                .map_err(|err| {
388                    ProviderError::UploadFile(location.to_string_lossy().to_string(), err.into())
389                })?;
390        }
391
392        Ok((url, content_hashed))
393    }

407    async fn download_file(
408        &self,
409        url: &str,
410        remote_file_path: &Path,
411        hash: Option<&str>,
412    ) -> Result<(), ProviderError> {
413        let r = self
414            .k8s_client
415            .pod_exec(
416                &self.namespace_name(),
417                &self.name,
418                vec![
419                    "/cfg/curl",
420                    url,
421                    "--output",
422                    &remote_file_path.to_string_lossy(),
423                ],
424            )
425            .await
426            .map_err(|err| {
427                ProviderError::DownloadFile(
428                    remote_file_path.to_string_lossy().to_string(),
429                    anyhow!(format!("node: {}, err: {}", self.name(), err)),
430                )
431            })?;
432
433        trace!("download url {} result: {:?}", url, r);
434
435        if r.is_err() {
436            return Err(ProviderError::DownloadFile(
437                remote_file_path.to_string_lossy().to_string(),
438                anyhow!(format!("node: {}, err downloading file", self.name())),
439            ));
440        }
441
442        if let Some(hash) = hash {
443            // check if the hash of the file is correct
444            let res = self
445                .k8s_client
446                .pod_exec(
447                    &self.namespace_name(),
448                    &self.name,
449                    vec![
450                        "/cfg/coreutils",
451                        "sha256sum",
452                        &remote_file_path.to_string_lossy(),
453                    ],
454                )
455                .await
456                .map_err(|err| {
457                    ProviderError::DownloadFile(
458                        remote_file_path.to_string_lossy().to_string(),
459                        anyhow!(format!("node: {}, err: {}", self.name(), err)),
460                    )
461                })?;
462
463            if let Ok(output) = res {
464                if !output.contains(hash) {
465                    return Err(ProviderError::DownloadFile(
466                        remote_file_path.to_string_lossy().to_string(),
467                        anyhow!(format!("node: {}, invalid sha256sum hash: {hash} for file, output was {output}", self.name())),
468                    ));
469                }
470            } else {
471                return Err(ProviderError::DownloadFile(
472                    remote_file_path.to_string_lossy().to_string(),
473                    anyhow!(format!(
474                        "node: {}, err calculating sha256sum for file {:?}",
475                        self.name(),
476                        res
477                    )),
478                ));
479            }
480        }
481
482        Ok(())
483    }

629    async fn send_file(
630        &self,
631        local_file_path: &Path,
632        remote_file_path: &Path,
633        mode: &str,
634    ) -> Result<(), ProviderError> {
635        if let Some(remote_parent_dir) = self.get_remote_parent_dir(remote_file_path) {
636            self.create_remote_dir(&remote_parent_dir).await?;
637        }
638
639        debug!(
640            "Uploading file: {} IFF not present in the fileserver",
641            local_file_path.to_string_lossy()
642        );
643
644        // we need to override the url to use inside the pod
645        let (mut url, hash) = self.upload_to_fileserver(local_file_path).await?;
646        let _ = url.set_host(Some("fileserver"));
647        let _ = url.set_port(Some(80));
648
649        // Sometimes downloading the file fails (the file is corrupted)
650        // Add at most 5 retries
651        let mut last_err = None;
652        for i in 0..5 {
653            if i > 0 {
654                warn!("retrying number {i} download file {:?}", remote_file_path);
655                tokio::time::sleep(Duration::from_secs(i)).await;
656            }
657
658            let res = self
659                .download_file(url.as_ref(), remote_file_path, Some(&hash))
660                .await;
661
662            last_err = res.err();
663
664            if last_err.is_none() {
665                // ready to continue
666                break;
667            }
668        }
669
670        if let Some(last_err) = last_err {
671            return Err(last_err);
672        }
673
674        let _ = self
675            .k8s_client
676            .pod_exec(
677                &self.namespace_name(),
678                &self.name,
679                vec!["chmod", mode, &remote_file_path.to_string_lossy()],
680            )
681            .await
682            .map_err(|err| {
683                ProviderError::SendFile(
684                    self.name.clone(),
685                    local_file_path.to_string_lossy().to_string(),
686                    err.into(),
687                )
688            })?;
689
690        Ok(())
691    }

733                    anyhow!("error when pausing node: status {}: {}", err.0, err.1),
734                )
735            })?;
736
737        Ok(())
738    }

740    async fn resume(&self) -> Result<(), ProviderError> {
741        self.k8s_client
742            .pod_exec(
743                &self.namespace_name(),
744                &self.name,
745                vec!["sh", "-c", "echo resume > /tmp/zombiepipe"],
746            )
747            .await
748            .map_err(|err| ProviderError::ResumeNodeFailed(self.name.to_string(), err.into()))?
749            .map_err(|err| {
750                ProviderError::ResumeNodeFailed(
751                    self.name.to_string(),
752                    anyhow!("error when pausing node: status {}: {}", err.0, err.1),
753                )
754            })?;
755
756        Ok(())
757    }

759    async fn restart(&self, after: Option<Duration>) -> Result<(), ProviderError> {
760        if let Some(duration) = after {
761            sleep(duration).await;
762        }
763
764        self.k8s_client
765            .pod_exec(
766                &self.namespace_name(),
767                &self.name,
768                vec!["sh", "-c", "echo restart > /tmp/zombiepipe"],
769            )
770            .await
771            .map_err(|err| ProviderError::RestartNodeFailed(self.name.to_string(), err.into()))?
772            .map_err(|err| {
773                ProviderError::RestartNodeFailed(
774                    self.name.to_string(),
775                    anyhow!("error when restarting node: status {}: {}", err.0, err.1),
776                )
777            })?;
778
779        Ok(())
780    }

782    async fn destroy(&self) -> Result<(), ProviderError> {
783        self.k8s_client
784            .delete_pod(&self.namespace_name(), &self.name)
785            .await
786            .map_err(|err| ProviderError::KillNodeFailed(self.name.to_string(), err.into()))?;
787
788        if let Some(namespace) = self.namespace.upgrade() {
789            namespace.nodes.write().await.remove(&self.name);
790        }
791
792        Ok(())
793    }
