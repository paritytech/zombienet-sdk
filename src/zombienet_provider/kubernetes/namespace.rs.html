58    pub(super) async fn new(
59        provider: &Weak<KubernetesProvider<FS>>,
60        tmp_dir: &PathBuf,
61        capabilities: &ProviderCapabilities,
62        k8s_client: &KubernetesClient,
63        filesystem: &FS,
64        custom_base_dir: Option<&Path>,
65    ) -> Result<Arc<Self>, ProviderError> {
66        // If the namespace is already provided
67        let name = if let Ok(name) = env::var(ZOMBIE_K8S_CI_NAMESPACE) {
68            name
69        } else {
70            format!("{}{}", NAMESPACE_PREFIX, Uuid::new_v4())
71        };
72
73        let base_dir = if let Some(custom_base_dir) = custom_base_dir {
74            if !filesystem.exists(custom_base_dir).await {
75                filesystem.create_dir(custom_base_dir).await?;
76            } else {
77                warn!(
78                    "⚠️ Using and existing directory {} as base dir",
79                    custom_base_dir.to_string_lossy()
80                );
81            }
82            PathBuf::from(custom_base_dir)
83        } else {
84            let base_dir = PathBuf::from_iter([tmp_dir, &PathBuf::from(&name)]);
85            filesystem.create_dir(&base_dir).await?;
86            base_dir
87        };
88
89        let namespace = Arc::new_cyclic(|weak| KubernetesNamespace {
90            weak: weak.clone(),
91            provider: provider.clone(),
92            name,
93            base_dir,
94            capabilities: capabilities.clone(),
95            filesystem: filesystem.clone(),
96            k8s_client: k8s_client.clone(),
97            file_server_port: RwLock::new(None),
98            file_server_fw_task: RwLock::new(None),
99            nodes: RwLock::new(HashMap::new()),
100            delete_on_drop: Arc::new(Mutex::new(true)),
101        });
102
103        namespace.initialize().await?;
104
105        Ok(namespace)
106    }

143    async fn initialize_k8s(&self) -> Result<(), ProviderError> {
144        // TODO (javier): check with Hamid if we are using this labels in any scheduling logic.
145        let labels = BTreeMap::from([
146            (
147                "jobId".to_string(),
148                env::var("CI_JOB_ID").unwrap_or("".to_string()),
149            ),
150            (
151                "projectName".to_string(),
152                env::var("CI_PROJECT_NAME").unwrap_or("".to_string()),
153            ),
154            (
155                "projectId".to_string(),
156                env::var("CI_PROJECT_ID").unwrap_or("".to_string()),
157            ),
158        ]);
159
160        let manifest = self
161            .k8s_client
162            .create_namespace(&self.name, labels)
163            .await
164            .map_err(|err| {
165                ProviderError::CreateNamespaceFailed(self.name.to_string(), err.into())
166            })?;
167
168        let serialized_manifest = serde_yaml::to_string(&manifest).map_err(|err| {
169            ProviderError::CreateNamespaceFailed(self.name.to_string(), err.into())
170        })?;
171
172        let dest_path =
173            PathBuf::from_iter([&self.base_dir, &PathBuf::from("namespace_manifest.yaml")]);
174
175        self.filesystem
176            .write(dest_path, serialized_manifest)
177            .await?;
178
179        Ok(())
180    }

209    async fn initialize_file_server(&self) -> Result<(), ProviderError> {
210        let name = "fileserver".to_string();
211        let labels = BTreeMap::from([
212            ("app.kubernetes.io/name".to_string(), name.clone()),
213            (
214                "x-infra-instance".to_string(),
215                env::var("X_INFRA_INSTANCE").unwrap_or("ondemand".to_string()),
216            ),
217        ]);
218
219        let pod_spec = PodSpec {
220            hostname: Some(name.clone()),
221            containers: vec![Container {
222                name: name.clone(),
223                image: Some(FILE_SERVER_IMAGE.to_string()),
224                image_pull_policy: Some("Always".to_string()),
225                ports: Some(vec![ContainerPort {
226                    container_port: 80,
227                    ..Default::default()
228                }]),
229                startup_probe: Some(Probe {
230                    http_get: Some(HTTPGetAction {
231                        path: Some("/".to_string()),
232                        port: IntOrString::Int(80),
233                        ..Default::default()
234                    }),
235                    initial_delay_seconds: Some(1),
236                    period_seconds: Some(2),
237                    failure_threshold: Some(3),
238                    ..Default::default()
239                }),
240                ..Default::default()
241            }],
242            restart_policy: Some("OnFailure".into()),
243            ..Default::default()
244        };
245
246        let pod_manifest = self
247            .k8s_client
248            .create_pod(&self.name, &name, pod_spec, labels.clone())
249            .await
250            .map_err(|err| ProviderError::FileServerSetupError(err.into()))?;
251
252        // TODO: remove duplication across methods
253        let pod_serialized_manifest = serde_yaml::to_string(&pod_manifest)
254            .map_err(|err| ProviderError::FileServerSetupError(err.into()))?;
255
256        let pod_dest_path = PathBuf::from_iter([
257            &self.base_dir,
258            &PathBuf::from("file_server_pod_manifest.yaml"),
259        ]);
260
261        self.filesystem
262            .write(pod_dest_path, pod_serialized_manifest)
263            .await?;
264
265        let service_spec = ServiceSpec {
266            selector: Some(labels.clone()),
267            ports: Some(vec![ServicePort {
268                port: 80,
269                ..Default::default()
270            }]),
271            ..Default::default()
272        };
273
274        let service_manifest = self
275            .k8s_client
276            .create_service(&self.name, &name, service_spec, labels)
277            .await
278            .map_err(|err| ProviderError::FileServerSetupError(err.into()))?;
279
280        let serialized_service_manifest = serde_yaml::to_string(&service_manifest)
281            .map_err(|err| ProviderError::FileServerSetupError(err.into()))?;
282
283        let service_dest_path = PathBuf::from_iter([
284            &self.base_dir,
285            &PathBuf::from("file_server_service_manifest.yaml"),
286        ]);
287
288        self.filesystem
289            .write(service_dest_path, serialized_service_manifest)
290            .await?;
291
292        let (port, task) = self
293            .k8s_client
294            .create_pod_port_forward(&self.name, &name, 0, 80)
295            .await
296            .map_err(|err| ProviderError::FileServerSetupError(err.into()))?;
297
298        *self.file_server_port.write().await = Some(port);
299        *self.file_server_fw_task.write().await = Some(task);
300
301        Ok(())
302    }

342    pub async fn delete_on_drop(&self) -> bool {
343        if let Ok(delete_on_drop) = self.delete_on_drop.try_lock() {
344            *delete_on_drop
345        } else {
346            // if we can't lock just remove the ns

352impl<FS> Drop for KubernetesNamespace<FS>
353where
354    FS: FileSystem + Send + Sync + Clone,
355{
356    fn drop(&mut self) {
357        let ns_name = self.name.clone();
358        if let Ok(delete_on_drop) = self.delete_on_drop.try_lock() {
359            if *delete_on_drop {
360                let client = self.k8s_client.clone();
361                let provider = self.provider.upgrade();
362                futures::executor::block_on(async move {
363                    trace!("🧟 deleting ns {ns_name} from cluster");
364                    let _ = client.delete_namespace(&ns_name).await;
365                    if let Some(provider) = provider {
366                        provider.namespaces.write().await.remove(&ns_name);
367                    }
368
369                    trace!("✅ deleted");
370                });
371            } else {
372                trace!("⚠️ leaking ns {ns_name} in cluster");
373            }
374        };
375    }

412    async fn get_node_available_args(
413        &self,
414        (command, image): (String, Option<String>),
415    ) -> Result<String, ProviderError> {
416        let node_image = image.expect(&format!("image should be present when getting node available args with kubernetes provider {THIS_IS_A_BUG}"));
417
418        // run dummy command in new pod
419        let temp_node = self
420            .spawn_node(
421                &SpawnNodeOptions::new(format!("temp-{}", Uuid::new_v4()), "cat".to_string())
422                    .image(node_image.clone()),
423            )
424            .await?;
425
426        let available_args_output = temp_node
427            .run_command(RunCommandOptions::new(command.clone()).args(vec!["--help"]))
428            .await?
429            .map_err(|(_exit, status)| {
430                ProviderError::NodeAvailableArgsError(node_image, command, status)
431            })?;
432
433        temp_node.destroy().await?;
434
435        Ok(available_args_output)
436    }

438    async fn spawn_node(&self, options: &SpawnNodeOptions) -> Result<DynNode, ProviderError> {
439        trace!("spawn node options {options:?}");

444            name: &options.name,

455        .await?;

465    async fn generate_files(&self, options: GenerateFilesOptions) -> Result<(), ProviderError> {
466        debug!("generate files options {options:#?}");
467
468        let node_name = options
469            .temp_name
470            .unwrap_or_else(|| format!("temp-{}", Uuid::new_v4()));
471        let node_image = options
472            .image
473            .expect(&format!("image should be present when generating files with kubernetes provider {THIS_IS_A_BUG}"));
474
475        // run dummy command in new pod
476        let temp_node = self
477            .spawn_node(
478                &SpawnNodeOptions::new(node_name, "cat".to_string())
479                    .injected_files(options.injected_files)
480                    .image(node_image),
481            )
482            .await?;
483
484        for GenerateFileCommand {
485            program,
486            args,
487            env,
488            local_output_path,
489        } in options.commands
490        {
491            let local_output_full_path = format!(
492                "{}{}{}",
493                self.base_dir.to_string_lossy(),
494                if local_output_path.starts_with("/") {
495                    ""
496                } else {
497                    "/"
498                },
499                local_output_path.to_string_lossy()
500            );
501
502            let contents = extract_execution_result(
503                &temp_node,
504                RunCommandOptions { program, args, env },
505                options.expected_path.as_ref(),
506            )
507            .await?;
508            self.filesystem
509                .write(local_output_full_path, contents)
510                .await
511                .map_err(|err| ProviderError::FileGenerationFailed(err.into()))?;
512        }
513
514        temp_node.destroy().await
515    }
