3/// Parse args to extract those marked for removal (with `-:` prefix).
4/// Returns a set of arg names/flags that should be removed from the final command.
5///
6/// # Examples
7/// - `-:--insecure-validator-i-know-what-i-do` -> removes `--insecure-validator-i-know-what-i-do`
8/// - `-:insecure-validator` -> removes `--insecure-validator` (normalized)
9/// - `-:--prometheus-port` -> removes `--prometheus-port`
10pub fn parse_removal_args(args: &[Arg]) -> Vec<String> {
11    args.iter()
12        .filter_map(|arg| match arg {
13            Arg::Flag(flag) if flag.starts_with("-:") => {
14                let mut flag_to_exclude = flag[2..].to_string();
15
16                // Normalize flag format - ensure it starts with --
17                if !flag_to_exclude.starts_with("--") {
18                    flag_to_exclude = format!("--{flag_to_exclude}");
19                }
20
21                Some(flag_to_exclude)
22            },
23            _ => None,
24        })
25        .collect()
26}
27
28/// Apply arg removals to a vector of string arguments.
29/// This filters out any args that match the removal list.
30///
31/// # Arguments
32/// * `args` - The command arguments to filter
33/// * `removals` - List of arg names/flags to remove
34///
35/// # Returns
36/// Filtered vector with specified args removed
37pub fn apply_arg_removals(args: Vec<String>, removals: &[String]) -> Vec<String> {
38    if removals.is_empty() {
39        return args;
40    }
41
42    let mut res = Vec::new();
43    let mut skip_next = false;
44
45    for (i, arg) in args.iter().enumerate() {
46        if skip_next {
47            skip_next = false;
48            continue;
49        }
50
51        let should_remove = removals
52            .iter()
53            .any(|removal| arg == removal || arg.starts_with(&format!("{removal}=")));
54
55        if should_remove {
56            // Only skip next if this looks like an option (starts with --) and next arg doesn't start with --
57            if !arg.contains("=") && i + 1 < args.len() {
58                let next_arg = &args[i + 1];
59                if !next_arg.starts_with("-") {
60                    skip_next = true;
61                }
62            }
63            continue;
64        }
65
66        res.push(arg.clone());
67    }
68
69    res
70}

72#[cfg(test)]
73mod tests {
74    use super::*;

76    #[test]
77    fn test_parse_removal_args() {
78        let args = vec![
79            Arg::Flag("-:--insecure-validator-i-know-what-i-do".to_string()),
80            Arg::Flag("--validator".to_string()),
81            Arg::Flag("-:--no-telemetry".to_string()),
82        ];
83
84        let removals = parse_removal_args(&args);
85        assert_eq!(removals.len(), 2);
86        assert!(removals.contains(&"--insecure-validator-i-know-what-i-do".to_string()));
87        assert!(removals.contains(&"--no-telemetry".to_string()));
88    }
89
90    #[test]
91    fn test_apply_arg_removals_flag() {
92        let args = vec![
93            "--validator".to_string(),
94            "--insecure-validator-i-know-what-i-do".to_string(),
95            "--no-telemetry".to_string(),
96        ];
97        let removals = vec!["--insecure-validator-i-know-what-i-do".to_string()];
98        let res = apply_arg_removals(args, &removals);
99        assert_eq!(res.len(), 2);
100        assert!(res.contains(&"--validator".to_string()));
101        assert!(res.contains(&"--no-telemetry".to_string()));
102        assert!(!res.contains(&"--insecure-validator-i-know-what-i-do".to_string()));
103    }

105    #[test]
