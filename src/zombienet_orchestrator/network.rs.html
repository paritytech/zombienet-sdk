131        &mut self,
132        name: impl Into<String>,

143            chain_spec_custom_path.clone()
144        } else {
145            PathBuf::from(format!(
146                "{}/{}.json",
147                self.ns.base_dir().to_string_lossy(),
148                relaychain.chain
149            ))
150        };
151
152        let chain_context = ChainDefaultContext {
153            default_command: self.initial_spec.relaychain.default_command.as_ref(),

160        let mut node_spec = network_spec::node::NodeSpec::from_ad_hoc(
161            &name,

165        )?;

170                .await?,

172

175

178            parachain_id: None,
179            chain: &relaychain.chain,

236    ///     ..Default::default()
237    /// };

239    /// network.add_collator("new-col-1", col_opts, 100).await?;

243    pub async fn add_collator(
244        &mut self,
245        name: impl Into<String>,
246        options: AddCollatorOptions,
247        para_id: u32,
248    ) -> Result<(), anyhow::Error> {
249        let name = generate_unique_node_name_from_names(
250            name,
251            &mut self.nodes_by_name.keys().cloned().collect(),
252        );
253        let spec = self
254            .initial_spec
255            .parachains
256            .iter()
257            .find(|para| para.id == para_id)
258            .ok_or(anyhow::anyhow!(format!("parachain: {para_id} not found!")))?;
259        let role = if spec.is_cumulus_based {
260            ZombieRole::CumulusCollator
261        } else {
262            ZombieRole::Collator
263        };
264        let chain_context = ChainDefaultContext {
265            default_command: spec.default_command.as_ref(),
266            default_image: spec.default_image.as_ref(),
267            default_resources: spec.default_resources.as_ref(),
268            default_db_snapshot: spec.default_db_snapshot.as_ref(),
269            default_args: spec.default_args.iter().collect(),
270        };
271
272        let parachain = self
273            .parachains
274            .get_mut(&para_id)
275            .ok_or(anyhow::anyhow!(format!("parachain: {para_id} not found!")))?
276            .get_mut(0)
277            .ok_or(anyhow::anyhow!(format!("parachain: {para_id} not found!")))?;
278
279        let base_dir = self.ns.base_dir().to_string_lossy();
280        let scoped_fs = ScopedFilesystem::new(&self.filesystem, &base_dir);
281
282        // TODO: we want to still supporting spawn a dedicated bootnode??
283        let ctx = SpawnNodeCtx {
284            chain_id: &self.relay.chain_id,
285            parachain_id: parachain.chain_id.as_deref(),
286            chain: &self.relay.chain,
287            role,
288            ns: &self.ns,
289            scoped_fs: &scoped_fs,
290            parachain: Some(spec),
291            bootnodes_addr: &vec![],
292            wait_ready: true,
293            nodes_by_name: serde_json::to_value(&self.nodes_by_name)?,
294            global_settings: &self.initial_spec.global_settings,
295        };
296
297        let relaychain_spec_path = if let Some(chain_spec_custom_path) = &options.chain_spec_relay {
298            chain_spec_custom_path.clone()
299        } else {
300            PathBuf::from(format!(
301                "{}/{}.json",
302                self.ns.base_dir().to_string_lossy(),
303                self.relay.chain
304            ))
305        };
306
307        let mut global_files_to_inject = vec![TransferedFile::new(
308            relaychain_spec_path,
309            PathBuf::from(format!("/cfg/{}.json", self.relay.chain)),
310        )];
311
312        let para_chain_spec_local_path = if let Some(para_chain_spec_custom) = &options.chain_spec {
313            Some(para_chain_spec_custom.clone())
314        } else if let Some(para_spec_path) = &parachain.chain_spec_path {
315            Some(PathBuf::from(format!(
316                "{}/{}",
317                self.ns.base_dir().to_string_lossy(),
318                para_spec_path.to_string_lossy()
319            )))
320        } else {
321            None
322        };
323
324        if let Some(para_spec_path) = para_chain_spec_local_path {
325            global_files_to_inject.push(TransferedFile::new(
326                para_spec_path,
327                PathBuf::from(format!("/cfg/{para_id}.json")),
328            ));
329        }
330
331        let mut node_spec =
332            network_spec::node::NodeSpec::from_ad_hoc(name, options.into(), &chain_context, true)?;
333
334        node_spec.available_args_output = Some(
335            self.initial_spec
336                .node_available_args_output(&node_spec, self.ns.clone())
337                .await?,
338        );
339
340        let node = spawner::spawn_node(&node_spec, global_files_to_inject, &ctx).await?;
341
342        // Let's make sure node is up before adding
343        node.wait_until_is_up(self.initial_spec.global_settings.network_spawn_timeout())
344            .await?;
345
346        parachain.collators.push(node.clone());
347        self.add_running_node(node, None).await;
348
349        Ok(())
350    }

353    ///
354    /// This allow you to build a new parachain config to be deployed into
355    /// the running network.
356    pub fn para_config_builder(&self) -> ParachainConfigBuilder<Initial, Running> {
357        let used_ports = self
358            .nodes_iter()
359            .map(|node| node.spec())
360            .flat_map(|spec| {
361                [
362                    spec.ws_port.0,
363                    spec.rpc_port.0,
364                    spec.prometheus_port.0,
365                    spec.p2p_port.0,
366                ]
367            })
368            .collect();
369
370        let used_nodes_names = self.nodes_by_name.keys().cloned().collect();
371
372        // need to inverse logic of generate_unique_para_id
373        let used_para_ids = self
374            .parachains
375            .iter()
376            .map(|(id, paras)| (*id, paras.len().saturating_sub(1) as u8))
377            .collect();
378
379        let context = ValidationContext {
380            used_ports,
381            used_nodes_names,
382            used_para_ids,
383        };
384        let context = Rc::new(RefCell::new(context));
385
386        ParachainConfigBuilder::new_with_running(context)
387    }

393    /// * `custom_relaychain_spec` - Optional path to a custom relaychain spec to use

397    /// # Example:
398    /// ```rust
399    /// # use anyhow::anyhow;
400    /// # use provider::NativeProvider;
401    /// # use support::{fs::local::LocalFileSystem};
402    /// # use zombienet_orchestrator::{errors, AddCollatorOptions, Orchestrator};
403    /// # use configuration::NetworkConfig;
404    /// # async fn example() -> Result<(), anyhow::Error> {
405    /// #   let provider = NativeProvider::new(LocalFileSystem {});
406    /// #   let orchestrator = Orchestrator::new(LocalFileSystem {}, provider);
407    /// #   let config = NetworkConfig::load_from_toml("config.toml")?;
408    /// let mut network = orchestrator.spawn(config).await?;
409    /// let para_config = network
410    ///     .para_config_builder()
411    ///     .with_id(100)
412    ///     .with_default_command("polkadot-parachain")
413    ///     .with_collator(|c| c.with_name("col-100-1"))
414    ///     .build()
415    ///     .map_err(|_e| anyhow!("Building config"))?;
416    ///
417    /// network.add_parachain(&para_config, None, None).await?;
418    ///
419    /// #   Ok(())
420    /// # }
421    /// ```
422    pub async fn add_parachain(
423        &mut self,
424        para_config: &ParachainConfig,
425        custom_relaychain_spec: Option<PathBuf>,
426        custom_parchain_fs_prefix: Option<String>,
427    ) -> Result<(), anyhow::Error> {
428        let base_dir = self.ns.base_dir().to_string_lossy().to_string();
429        let scoped_fs = ScopedFilesystem::new(&self.filesystem, &base_dir);
430
431        let mut global_files_to_inject = vec![];
432
433        // get relaychain id
434        let relay_chain_id = if let Some(custom_path) = custom_relaychain_spec {
435            // use this file as relaychain spec
436            global_files_to_inject.push(TransferedFile::new(
437                custom_path.clone(),
438                PathBuf::from(format!("/cfg/{}.json", self.relaychain().chain)),
439            ));
440            let content = std::fs::read_to_string(custom_path)?;
441            ChainSpec::chain_id_from_spec(&content)?
442        } else {
443            global_files_to_inject.push(TransferedFile::new(
444                PathBuf::from(format!(
445                    "{}/{}",
446                    scoped_fs.base_dir,
447                    self.relaychain().chain_spec_path.to_string_lossy()
448                )),
449                PathBuf::from(format!("/cfg/{}.json", self.relaychain().chain)),
450            ));
451            self.relay.chain_id.clone()
452        };
453
454        let mut para_spec = network_spec::parachain::ParachainSpec::from_config(
455            para_config,
456            relay_chain_id.as_str().try_into()?,
457        )?;
458
459        let chain_spec_raw_path = para_spec
460            .build_chain_spec(&relay_chain_id, &self.ns, &scoped_fs)
461            .await?;
462
463        // Para artifacts
464        let para_path_prefix = if let Some(custom_prefix) = custom_parchain_fs_prefix {
465            custom_prefix
466        } else {
467            para_spec.id.to_string()
468        };
469
470        scoped_fs.create_dir(&para_path_prefix).await?;
471        // create wasm/state
472        para_spec
473            .genesis_state
474            .build(
475                chain_spec_raw_path.as_ref(),
476                format!("{}/genesis-state", &para_path_prefix),
477                &self.ns,
478                &scoped_fs,
479                None,
480            )
481            .await?;
482        para_spec
483            .genesis_wasm
484            .build(
485                chain_spec_raw_path.as_ref(),
486                format!("{}/para_spec-wasm", &para_path_prefix),
487                &self.ns,
488                &scoped_fs,
489                None,
490            )
491            .await?;
492
493        let parachain =
494            Parachain::from_spec(&para_spec, &global_files_to_inject, &scoped_fs).await?;
495        let parachain_id = parachain.chain_id.clone();
496
497        // Create `ctx` for spawn the nodes
498        let ctx_para = SpawnNodeCtx {
499            parachain: Some(&para_spec),
500            parachain_id: parachain_id.as_deref(),
501            role: if para_spec.is_cumulus_based {
502                ZombieRole::CumulusCollator
503            } else {
504                ZombieRole::Collator
505            },
506            bootnodes_addr: &para_config
507                .bootnodes_addresses()
508                .iter()
509                .map(|&a| a.to_string())
510                .collect(),
511            chain_id: &self.relaychain().chain_id,
512            chain: &self.relaychain().chain,
513            ns: &self.ns,
514            scoped_fs: &scoped_fs,
515            wait_ready: false,
516            nodes_by_name: serde_json::to_value(&self.nodes_by_name)?,
517            global_settings: &self.initial_spec.global_settings,
518        };
519
520        // Register the parachain to the running network
521        let first_node_url = self
522            .relaychain()
523            .nodes
524            .first()
525            .ok_or(anyhow::anyhow!(
526                "At least one node of the relaychain should be running"
527            ))?
528            .ws_uri();
529
530        if para_config.registration_strategy() == Some(&RegistrationStrategy::UsingExtrinsic) {
531            let register_para_options = RegisterParachainOptions {
532                id: parachain.para_id,
533                // This needs to resolve correctly
534                wasm_path: para_spec
535                    .genesis_wasm
536                    .artifact_path()
537                    .ok_or(anyhow::anyhow!(
538                        "artifact path for wasm must be set at this point",
539                    ))?
540                    .to_path_buf(),
541                state_path: para_spec
542                    .genesis_state
543                    .artifact_path()
544                    .ok_or(anyhow::anyhow!(
545                        "artifact path for state must be set at this point",
546                    ))?
547                    .to_path_buf(),
548                node_ws_url: first_node_url.to_string(),
549                onboard_as_para: para_spec.onboard_as_parachain,
550                seed: None, // TODO: Seed is passed by?
551                finalization: false,
552            };
553
554            Parachain::register(register_para_options, &scoped_fs).await?;
555        }
556
557        // Spawn the nodes
558        let spawning_tasks = para_spec
559            .collators
560            .iter()
561            .map(|node| spawner::spawn_node(node, parachain.files_to_inject.clone(), &ctx_para));
562
563        let running_nodes = futures::future::try_join_all(spawning_tasks).await?;
564
565        // Let's make sure nodes are up before adding them
566        let waiting_tasks = running_nodes.iter().map(|node| {
567            node.wait_until_is_up(self.initial_spec.global_settings.network_spawn_timeout())
568        });
569
570        let _ = futures::future::try_join_all(waiting_tasks).await?;
571
572        let running_para_id = parachain.para_id;
573        self.add_para(parachain);
574        for node in running_nodes {
575            self.add_running_node(node, Some(running_para_id)).await;
576        }
577
578        Ok(())
579    }

619    pub async fn register_parachain(&mut self, para_id: u32) -> Result<(), anyhow::Error> {
620        let para = self
621            .initial_spec
622            .parachains
623            .iter()
624            .find(|p| p.id == para_id)
625            .ok_or(anyhow::anyhow!(
626                "no parachain with id = {para_id} available",
627            ))?;
628        let para_genesis_config = para.get_genesis_config()?;
629        let first_node_url = self
630            .relaychain()
631            .nodes
632            .first()
633            .ok_or(anyhow::anyhow!(
634                "At least one node of the relaychain should be running"
635            ))?
636            .ws_uri();
637        let register_para_options: RegisterParachainOptions = RegisterParachainOptions {
638            id: para_id,
639            // This needs to resolve correctly
640            wasm_path: para_genesis_config.wasm_path.clone(),
641            state_path: para_genesis_config.state_path.clone(),
642            node_ws_url: first_node_url.to_string(),
643            onboard_as_para: para_genesis_config.as_parachain,
644            seed: None, // TODO: Seed is passed by?
645            finalization: false,
646        };
647        let base_dir = self.ns.base_dir().to_string_lossy().to_string();
648        let scoped_fs = ScopedFilesystem::new(&self.filesystem, &base_dir);
649        Parachain::register(register_para_options, &scoped_fs).await?;
650
651        Ok(())
652    }

671            "can't find node with name: {name:?}, should be one of {list}"

694    pub(crate) async fn add_running_node(&mut self, node: NetworkNode, para_id: Option<u32>) {
695        if let Some(para_id) = para_id {
696            if let Some(para) = self.parachains.get_mut(&para_id).and_then(|p| p.get_mut(0)) {
697                para.collators.push(node.clone());
698            } else {
699                // is the first node of the para, let create the entry
700                unreachable!()
701            }
702        } else {
703            self.relay.nodes.push(node.clone());
704        }
705        // TODO: we should hold a ref to the node in the vec in the future.
706        node.set_is_running(true);
707        let node_name = node.name.clone();
708        self.nodes_by_name.insert(node_name, node.clone());
709        self.nodes_to_watch.write().await.push(node);
710    }

769    /// Waits given number of seconds until all nodes in the network report that they are
770    /// up and running.
771    ///
772    /// # Arguments
773    /// * `timeout_secs` - The number of seconds to wait.
774    ///
775    /// # Returns
776    /// * `Ok()` if the node is up before timeout occured.
777    /// * `Err(e)` if timeout or other error occurred while waiting.
778    pub async fn wait_until_is_up(&self, timeout_secs: u64) -> Result<(), anyhow::Error> {
779        let handles = self
780            .nodes_iter()
781            .map(|node| node.wait_until_is_up(timeout_secs));
782
783        futures::future::try_join_all(handles).await?;
784
785        Ok(())
786    }

788    pub(crate) fn spawn_watching_task(&self) {
789        let nodes_to_watch = Arc::clone(&self.nodes_to_watch);
790        let ns = Arc::clone(&self.ns);
791
792        tokio::spawn(async move {
793            loop {
794                tokio::time::sleep(Duration::from_secs(NODE_MONITORING_INTERVAL_SECONDS)).await;
795
796                let all_running = {
797                    let guard = nodes_to_watch.read().await;
798                    let nodes = guard.iter().filter(|n| n.is_running()).collect::<Vec<_>>();
799
800                    let all_running =
801                        futures::future::try_join_all(nodes.iter().map(|n| {
802                            n.wait_until_is_up(NODE_MONITORING_FAILURE_THRESHOLD_SECONDS)
803                        }))
804                        .await;
805
806                    // Re-check `is_running` to make sure we don't kill the network unnecessarily
807                    if nodes.iter().any(|n| !n.is_running()) {
808                        continue;
809                    } else {
810                        all_running
811                    }
812                };
813
814                if let Err(e) = all_running {
815                    warn!("\n\t🧟 One of the nodes crashed: {e}. tearing the network down...");
816
817                    if let Err(e) = ns.destroy().await {
818                        error!("an error occurred during network teardown: {}", e);
819                    }
820
821                    std::process::exit(1);
822                }
823            }
824        });
825    }
