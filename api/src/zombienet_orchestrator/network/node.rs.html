66/// Result of waiting for a certain number of log lines to appear.
67///
68/// Indicates whether the log line count condition was met within the timeout period.
69///
70/// # Variants
71/// - `TargetReached(count)` â€“ The predicate condition was satisfied within the timeout.
72///     * `count`: The number of matching log lines at the time of satisfaction.
73/// - `TargetFailed(count)` â€“ The condition was not met within the timeout.
74///     * `count`: The final number of matching log lines at timeout expiration.
75#[derive(Debug, Clone, Copy, PartialEq, Eq)]
76pub enum LogLineCount {
77    TargetReached(u32),
78    TargetFailed(u32),
79}
80
81impl LogLineCount {
82    pub fn success(&self) -> bool {
83        match self {
84            Self::TargetReached(..) => true,
85            Self::TargetFailed(..) => false,
86        }
87    }
88}

90/// Configuration for controlling log line count waiting behavior.
91///
92/// Allows specifying a custom predicate on the number of matching log lines,
93/// a timeout in seconds, and whether the system should wait the entire timeout duration.
94///
95/// # Fields
96/// - `predicate`: A function that takes the current number of matching lines and
97///   returns `true` if the condition is satisfied.
98/// - `timeout_secs`: Maximum number of seconds to wait.
99/// - `wait_until_timeout_elapses`: If `true`, the system will continue waiting
100///   for the full timeout duration, even if the condition is already met early.
101///   Useful when you need to verify sustained absence or stability (e.g., "ensure no new logs appear").
102#[derive(Clone)]
103pub struct LogLineCountOptions {
104    pub predicate: Arc<dyn Fn(u32) -> bool + Send + Sync>,
105    pub timeout: Duration,
106    pub wait_until_timeout_elapses: bool,
107}
108
109impl LogLineCountOptions {
110    pub fn new(
111        predicate: impl Fn(u32) -> bool + 'static + Send + Sync,
112        timeout: Duration,
113        wait_until_timeout_elapses: bool,
114    ) -> Self {
115        Self {
116            predicate: Arc::new(predicate),
117            timeout,
118            wait_until_timeout_elapses,
119        }
120    }
121
122    pub fn no_occurences_within_timeout(timeout: Duration) -> Self {
123        Self::new(|n| n == 0, timeout, true)
124    }
125
126    pub fn at_least_once(timeout: Duration) -> Self {
127        Self::new(|count| count >= 1, timeout, false)
128    }
129
130    pub fn exactly_once(timeout: Duration) -> Self {
131        Self::new(|count| count == 1, timeout, false)
132    }

372    /// Wait until a metric value pass the `predicate`
373    pub async fn wait_metric(
374        &self,
375        metric_name: impl Into<String>,
376        predicate: impl Fn(f64) -> bool,
377    ) -> Result<(), anyhow::Error> {
378        let metric_name = metric_name.into();
379        trace!(
380            "[{}] waiting until metric {metric_name} pass the predicate",
381            self.name()
382        );
383        loop {
384            let res = self.assert_with(&metric_name, &predicate).await;
385            trace!("res: {res:?}");
386            match res {
387                Ok(res) => {
388                    if res {
389                        return Ok(());
390                    }
391                },
392                Err(e) => match e.downcast::<reqwest::Error>() {
393                    Ok(io_err) => {
394                        if !skip_err_while_waiting(&io_err) {
395                            return Err(io_err.into());
396                        }
397                    },
398                    Err(other) => {
399                        match other.downcast::<NetworkNodeError>() {
400                            Ok(node_err) => {
401                                if !matches!(node_err, NetworkNodeError::MetricNotFound(_)) {
402                                    return Err(node_err.into());
403                                }
404                            },
405                            Err(other) => return Err(other),
406                        };
407                    },
408                },
409            }
410
411            // sleep to not spam prometheus
412            tokio::time::sleep(Duration::from_secs(1)).await;
413        }
414    }

416    /// Wait until a metric value pass the `predicate`
417    /// with a timeout (secs)
418    pub async fn wait_metric_with_timeout(
419        &self,
420        metric_name: impl Into<String>,
421        predicate: impl Fn(f64) -> bool,
422        timeout_secs: impl Into<u64>,
423    ) -> Result<(), anyhow::Error> {
424        let metric_name = metric_name.into();
425        let secs = timeout_secs.into();
426        let log_msg = format!(
427            "[{}] waiting until metric {metric_name} pass the predicate for {secs}s",
428            self.name()
429        );
430        if metric_name == PROCESS_START_TIME_METRIC {
431            trace!("{log_msg}");
432        } else {
433            debug!("{log_msg}");
434        }
435        let res = tokio::time::timeout(
436            Duration::from_secs(secs),
437            self.wait_metric(&metric_name, predicate),
438        )
439        .await;
440
441        if let Ok(inner_res) = res {
442            match inner_res {
443                Ok(_) => Ok(()),
444                Err(e) => Err(anyhow!("Error waiting for metric: {e}")),
445            }
446        } else {
447            // timeout
448            Err(anyhow!(
449                "Timeout ({secs}), waiting for metric {metric_name} pass the predicate"
450            ))
451        }
452    }

462    /// Wait until a the number of matching log lines is reach
463    pub async fn wait_log_line_count(
464        &self,
465        pattern: impl Into<String>,
466        is_glob: bool,
467        count: usize,
468    ) -> Result<(), anyhow::Error> {
469        let pattern = pattern.into();
470        let pattern_clone = pattern.clone();
471        debug!("waiting until we find pattern {pattern} {count} times");
472        let match_fn: BoxedClosure = if is_glob {
473            Box::new(move |line: &str| Ok(glob_match(&pattern, line)))
474        } else {
475            let re = Regex::new(&pattern)?;
476            Box::new(move |line: &str| re.is_match(line).map_err(|e| anyhow!(e.to_string())))
477        };
478
479        loop {
480            let mut q = 0_usize;
481            let logs = self.logs().await?;
482            for line in logs.lines() {
483                trace!("line is {line}");
484                if match_fn(line)? {
485                    trace!("pattern {pattern_clone} match in line {line}");
486                    q += 1;
487                    if q >= count {
488                        return Ok(());
489                    }
490                }
491            }
492
493            tokio::time::sleep(Duration::from_secs(2)).await;
494        }
495    }

497    /// Waits until the number of matching log lines satisfies a custom condition,
498    /// optionally waiting for the entire duration of the timeout.
499    ///
500    /// This method searches log lines for a given substring or glob pattern,
501    /// and evaluates the number of matching lines using a user-provided predicate function.
502    /// Optionally, it can wait for the full timeout duration to ensure the condition
503    /// holds consistently (e.g., for verifying absence of logs).
504    ///
505    /// # Arguments
506    /// * `substring` - The substring or pattern to match within log lines.
507    /// * `is_glob` - Whether to treat `substring` as a glob pattern (`true`) or a regex (`false`).
508    /// * `options` - Configuration for timeout, match count predicate, and full-duration waiting.
509    ///
510    /// # Returns
511    /// * `Ok(LogLineCount::TargetReached(n))` if the predicate was satisfied within the timeout,
512    /// * `Ok(LogLineCount::TargetFails(n))` if the predicate was not satisfied in time,
513    /// * `Err(e)` if an error occurred during log retrieval or matching.
514    ///
515    /// # Example
516    /// ```rust
517    /// # use std::{sync::Arc, time::Duration};
518    /// # use provider::NativeProvider;
519    /// # use support::{fs::local::LocalFileSystem};
520    /// # use zombienet_orchestrator::{Orchestrator, network::node::{NetworkNode, LogLineCountOptions}};
521    /// # use configuration::NetworkConfig;
522    /// # async fn example() -> Result<(), anyhow::Error> {
523    /// #   let provider = NativeProvider::new(LocalFileSystem {});
524    /// #   let orchestrator = Orchestrator::new(LocalFileSystem {}, provider);
525    /// #   let config = NetworkConfig::load_from_toml("config.toml")?;
526    /// #   let network = orchestrator.spawn(config).await?;
527    /// let node = network.get_node("alice")?;
528    /// // Wait (up to 10 seconds) until pattern occurs once
529    /// let options = LogLineCountOptions {
530    ///     predicate: Arc::new(|count| count == 1),
531    ///     timeout: Duration::from_secs(10),
532    ///     wait_until_timeout_elapses: false,
533    /// };
534    /// let result = node
535    ///     .wait_log_line_count_with_timeout("error", false, options)
536    ///     .await?;
537    /// #   Ok(())
538    /// # }
539    /// ```
540    pub async fn wait_log_line_count_with_timeout(
541        &self,
542        substring: impl Into<String>,
543        is_glob: bool,
544        options: LogLineCountOptions,
545    ) -> Result<LogLineCount, anyhow::Error> {
546        let substring = substring.into();
547        debug!(
548            "waiting until match lines count within {} seconds",
549            options.timeout.as_secs_f64()
550        );
551
552        let start = tokio::time::Instant::now();
553
554        let match_fn: BoxedClosure = if is_glob {
555            Box::new(move |line: &str| Ok(glob_match(&substring, line)))
556        } else {
557            let re = Regex::new(&substring)?;
558            Box::new(move |line: &str| re.is_match(line).map_err(|e| anyhow!(e.to_string())))
559        };
560
561        if options.wait_until_timeout_elapses {
562            tokio::time::sleep(options.timeout).await;
563        }
564
565        let mut q;
566        loop {
567            q = 0_u32;
568            let logs = self.logs().await?;
569            for line in logs.lines() {
570                if match_fn(line)? {
571                    q += 1;
572
573                    // If `wait_until_timeout_elapses` is set then check the condition just once at the
574                    // end after the whole log file is processed. This is to address the cases when the
575                    // predicate becomes true and false again.
576                    // eg. expected exactly 2 matching lines are expected but 3 are present
577                    if !options.wait_until_timeout_elapses && (options.predicate)(q) {
578                        return Ok(LogLineCount::TargetReached(q));
579                    }
580                }
581            }
582
583            if start.elapsed() >= options.timeout {
584                break;
585            }
586
587            tokio::time::sleep(Duration::from_secs(2)).await;
588        }
589
590        if (options.predicate)(q) {
591            Ok(LogLineCount::TargetReached(q))
592        } else {
593            Ok(LogLineCount::TargetFailed(q))
594        }
595    }

606    async fn metric(
607        &self,
608        metric_name: &str,
609        treat_not_found_as_zero: bool,
610    ) -> Result<f64, anyhow::Error> {
611        let mut metrics_map = self.metrics_cache.read().await;
612        if metrics_map.is_empty() {

616            metrics_map = self.metrics_cache.read().await;

622            Ok(0_f64)
623        } else {
624            Err(NetworkNodeError::MetricNotFound(metric_name.into()).into())
625        }

628    /// Fetches histogram buckets for a given metric from the Prometheus endpoint.
629    ///
630    /// This function retrieves histogram bucket data by parsing the Prometheus metrics
631    /// and calculating the count of observations in each bucket. It automatically appends
632    /// `_bucket` suffix to the metric name if not already present.
633    ///
634    /// # Arguments
635    /// * `metric_name` - The name of the histogram metric (with or without `_bucket` suffix)
636    /// * `label_filters` - Optional HashMap of label key-value pairs to filter metrics by
637    ///
638    /// # Returns
639    /// A HashMap where keys are the `le` bucket boundaries as strings,
640    /// and values are the count of observations in each bucket (calculated as delta from previous bucket).
641    ///
642    /// # Example
643    /// ```ignore
644    /// let buckets = node.get_histogram_buckets("polkadot_pvf_execution_time", None).await?;
645    /// // Returns: {"0.1": 5, "0.5": 10, "1.0": 3, "+Inf": 0}
646    /// ```
647    pub async fn get_histogram_buckets(
648        &self,
649        metric_name: impl AsRef<str>,
650        label_filters: Option<HashMap<String, String>>,
651    ) -> Result<HashMap<String, u64>, anyhow::Error> {
652        let metric_name = metric_name.as_ref();
653
654        // Fetch and parse metrics using the existing parser
655        let response = reqwest::get(&self.prometheus_uri).await?;
656        let metrics = prom_metrics_parser::parse(&response.text().await?)?;
657
658        // Ensure metric name has _bucket suffix
659        let resolved_metric_name = if metric_name.contains("_bucket") {
660            metric_name.to_string()
661        } else {
662            format!("{}_bucket", metric_name)
663        };
664
665        // First pass: collect all matching metrics with their label counts
666        // to identify which ones have the most complete label sets
667        // Each entry contains: (full_metric_key, parsed_labels_map, cumulative_count)
668        let mut metric_entries: Vec<(String, HashMap<String, String>, u64)> = Vec::new();
669
670        for (key, &value) in metrics.iter() {
671            if !key.starts_with(&resolved_metric_name) {
672                continue;
673            }
674
675            let remaining = &key[resolved_metric_name.len()..];
676
677            let labels_str = &remaining[1..remaining.len() - 1];
678            let parsed_labels = Self::parse_label_string(labels_str);
679
680            // Must have "le" label
681            if !parsed_labels.contains_key("le") {
682                continue;
683            }
684
685            // Check if label filters match
686            if let Some(ref filters) = label_filters {
687                let mut all_match = true;
688                for (filter_key, filter_value) in filters {
689                    if parsed_labels.get(filter_key) != Some(filter_value) {
690                        all_match = false;
691                        break;
692                    }
693                }
694                if !all_match {
695                    continue;
696                }
697            }
698
699            metric_entries.push((key.clone(), parsed_labels, value as u64));
700        }
701
702        // Find the maximum number of labels (excluding "le") across all entries
703        // This helps us identify the "fullest" version of each metric
704        let max_label_count = metric_entries
705            .iter()
706            .map(|(_, labels, _)| labels.iter().filter(|(k, _)| k.as_str() != "le").count())
707            .max()
708            .unwrap_or(0);
709
710        // Second pass: collect buckets, deduplicating and preferring entries with more labels
711        let mut raw_buckets: Vec<(String, u64)> = Vec::new();
712        let mut seen_le_values = HashSet::new();
713        let mut active_series: Option<Vec<(String, String)>> = None;
714
715        for (_, parsed_labels, value) in metric_entries {
716            let le_value = parsed_labels.get("le").unwrap().clone();
717
718            // Get non-"le" labels
719            let mut non_le_labels: Vec<(String, String)> = parsed_labels
720                .iter()
721                .filter(|(k, _)| k.as_str() != "le")
722                .map(|(k, v)| (k.clone(), v.clone()))
723                .collect();
724            non_le_labels.sort();
725
726            // Only process entries that have the maximum number of labels
727            // (this filters out the parser's duplicate keys with fewer labels)
728            if non_le_labels.len() < max_label_count {
729                continue;
730            }
731
732            // Detect series changes
733            if let Some(ref prev_series) = active_series {
734                if prev_series != &non_le_labels {
735                    if !raw_buckets.is_empty() {
736                        break; // Stop at first series change
737                    }
738                    active_series = Some(non_le_labels.clone());
739                    seen_le_values.clear();
740                }
741            } else {
742                active_series = Some(non_le_labels.clone());
743            }
744
745            // Deduplicate by le value within this series
746            if !seen_le_values.insert(le_value.clone()) {
747                continue;
748            }
749
750            trace!("{} le:{} {}", resolved_metric_name, &le_value, value);
751            raw_buckets.push((le_value, value));
752        }
753
754        // Sort buckets by their "le" values
755        raw_buckets.sort_by(|a, b| Self::compare_le_values(&a.0, &b.0));
756
757        // Calculate deltas between cumulative buckets
758        let mut buckets = HashMap::new();
759        let mut previous_value = 0_u64;
760        for (le, cumulative_count) in raw_buckets {
761            if cumulative_count < previous_value {
762                warn!(
763                    "Warning: bucket count decreased from {} to {} at le={}",
764                    previous_value, cumulative_count, le
765                );
766            }
767            let delta = cumulative_count.saturating_sub(previous_value);
768            buckets.insert(le, delta);
769            previous_value = cumulative_count;
770        }
771
772        Ok(buckets)
773    }
774
775    /// Parse label string from parsed metric key.
776    ///
777    /// Takes a label string in the format `key1="value1",key2="value2"`
778    /// and returns a HashMap of key-value pairs.
779    /// Handles commas inside quoted values correctly.
780    fn parse_label_string(labels_str: &str) -> HashMap<String, String> {
781        let mut labels = HashMap::new();
782        let mut current_key = String::new();
783        let mut current_value = String::new();
784        let mut in_value = false;
785        let mut in_quotes = false;
786
787        for ch in labels_str.chars() {
788            match ch {
789                '=' if !in_quotes && !in_value => {
790                    in_value = true;
791                },
792                '"' if in_value => {
793                    in_quotes = !in_quotes;
794                },
795                ',' if !in_quotes => {
796                    // End of key-value pair
797                    if !current_key.is_empty() {
798                        labels.insert(
799                            current_key.trim().to_string(),
800                            current_value.trim().to_string(),
801                        );
802                        current_key.clear();
803                        current_value.clear();
804                        in_value = false;
805                    }
806                },
807                _ => {
808                    if in_value {
809                        current_value.push(ch);
810                    } else {
811                        current_key.push(ch);
812                    }
813                },
814            }
815        }
816
817        // Insert last pair
818        if !current_key.is_empty() {
819            labels.insert(
820                current_key.trim().to_string(),
821                current_value.trim().to_string(),
822            );
823        }
824
825        labels
826    }

828    /// Compare two histogram bucket boundary values for sorting.
829    ///
830    /// Treats "+Inf" as the maximum value, otherwise compares numerically.
831    fn compare_le_values(a: &str, b: &str) -> std::cmp::Ordering {
832        use std::cmp::Ordering;
833
834        // Handle +Inf specially
835        match (a, b) {
836            ("+Inf", "+Inf") => Ordering::Equal,
837            ("+Inf", _) => Ordering::Greater,
838            (_, "+Inf") => Ordering::Less,
839            _ => {
840                // Try to parse as f64 for numeric comparison
841                match (a.parse::<f64>(), b.parse::<f64>()) {
842                    (Ok(a_val), Ok(b_val)) => a_val.partial_cmp(&b_val).unwrap_or(Ordering::Equal),
843                    // Fallback to string comparison if parsing fails
844                    _ => a.cmp(b),
845                }
846            },
847        }
848    }

850    /// Waits given number of seconds until node reports that it is up and running, which
851    /// is determined by metric 'process_start_time_seconds', which should appear,
852    /// when node finished booting up.
853    ///
854    ///
855    /// # Arguments
856    /// * `timeout_secs` - The number of seconds to wait.
857    ///
858    /// # Returns
859    /// * `Ok()` if the node is up before timeout occured.
860    /// * `Err(e)` if timeout or other error occurred while waiting.
861    pub async fn wait_until_is_up(
862        &self,
863        timeout_secs: impl Into<u64>,
864    ) -> Result<(), anyhow::Error> {
865        self.wait_metric_with_timeout(PROCESS_START_TIME_METRIC, |b| b >= 1.0, timeout_secs)
866            .await
867            .map_err(|err| anyhow::anyhow!("{}: {:?}", self.name(), err))
868    }
869}
